int query(int i, int j) {

void solve() {
    int n, m;
    cin >> n >> m;

    int sr = 1, er = n;
    int sc = 1, ec = m;

    while (ec - sc + 1 > 25) {
        int mr = sr + 1;
        int w = ec - sc + 1;
        int j1 = sc + (3 * w) / 16;
        int j2 = sc + w / 2;
        int j3 = sc + (13 * w) / 16;

        int x1 = query(mr, j1);
        if (x1 == 0) {
            cout << "! " << mr << ' ' << j1 << endl;
            return;
        }

        int x2 = query(mr, j2);
        if (x2 == 0) {
            cout << "! " << mr << ' ' << j2 << endl;
            return;
        }

        int x3 = query(mr, j3);
        if (x3 == 0) {
            cout << "! " << mr << ' ' << j3 << endl;
            return;
        }

        int max_x = max({x1, x2, x3});
        int thr = (3 * w) / 4;

        if (max_x > thr) {
            sr += 3;
            if (sr > er) sr = er;
        } else {
            int min_x = min({x1, x2, x3});
            int jm = (x1 == min_x) ? j1 : (x2 == min_x) ? j2 : j3;

            sc = max(sc, jm - min_x);
            ec = min(ec, jm + min_x);
        }
    }

    for (int r = sr; r <= er; ++r) {
        int l = sc, r_col = ec;
        while (l <= r_col) {
            if (l == r_col) {
                int res = query(r, l);
                if (res == 0) {
                    cout << "! " << r << ' ' << l << endl;
                    return;
                }
                break;
            }

            int res = query(r, l);
            if (res == 0) {
                cout << "! " << r << ' ' << l << endl;
                return;
            }

            int new_l = max(l + 1, (res - 1 + 2 * l) / 2 + 1);
            int new_r = min(r_col, l + res);
            l = new_l;
            r_col = new_r;
        }
    }
}
